// PotholeReporter.swift
import Foundation
import CoreLocation

class PotholeReporter {
  static let baseURL = "http://YOUR_SERVER:8000"
  
  static func report(
    location: CLLocation,
    severity: String,
    confidence: Float,
    deviceId: String
  ) async throws -> [String: Any] {
    let url = URL(string:
      "\(baseURL)/api/potholes/ios")!
    var req = URLRequest(url: url)
    req.httpMethod = "POST"
    req.setValue("application/json",
      forHTTPHeaderField: "Content-Type")
    
    let body: [String: Any] = [
      "latitude": location.coordinate.latitude,
      "longitude": location.coordinate.longitude,
      "severity": severity,
      "description": "Auto-detected",
      "device_id": deviceId,
      "confidence": confidence,
      "model_version": "RDD2022-v3"
    ]
    
    req.httpBody = try
      JSONSerialization.data(
        withJSONObject: body)
    
    let (data, _) = try await
      URLSession.shared.data(for: req)
    
    return try JSONSerialization
      .jsonObject(with: data)
        as! [String: Any]
  }
}
